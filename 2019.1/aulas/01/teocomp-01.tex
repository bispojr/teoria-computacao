		\end{itemize}
	\end{block}

\begin{frame}{Instrumentos de Avaliação}
\begin{block}{Critérios de Aprovação}
	\begin{itemize}
		\item Estar presente em 75\% dos nossos encontros;
		\item Compromisso com a própria aprendizagem durante o semestre;
		\item Respeito com o professor e com os demais colegas;
		\item Apresentação de sua trajetória de formação para toda a turma.
	\end{itemize}
\end{block} \pause
